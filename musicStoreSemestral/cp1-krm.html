Relational model for musicstore database

  • User (nickname, email)
  • PrivateInfo (fullName, address, user, phoneNumber, password)
    • FK: (user) ⊆ User(nickname)
    • Address (phoneNumber, street, city, zip)
      • FK: Address(privateInfo) ⊆ PrivateInfo(phoneNumber)
  • WorkerUser (personalIdentificationNumber, nickname, email, location)
    • FK: WorkerUser(nickname, email) ⊆ User(nickname, email)
    • FK: WorkerUser(location) ⊆ Store(location)
  • superiority (isSuperior, isUnderling)
    • FK: superiority(isSuperior) ⊆ WorkerUser(personalIdentificationNumber)
    • FK: superiority(isUnderling) ⊆ WorkerUser(personalIdentificationNumber)
  • CustomerUser (nickname, email, cookies, premiumStatus)
    • FK: (nickname, email) ⊆ User(nickname, email)
  • buys (ISBN, nickname, email)
    • FK: buys(ISBN) ⊆ Product(ISBN)
    • FK: buys(nickname, email) ⊆ CustomerUser(nickname, email)
  • Store (location)
  • Product (ISBN, modelNumber, manufacturer, location)
    • FK: Product(location) ⊆ Store(location)
  • InstrumentProduct (ISBN, modelNumber, range, type, pickupType)
    • pickupType(ISBN)
      • FK: pickupType(ISBN) ⊆ InstrumentProduct(ISBN)
    • FK: InstrumentProduct(ISBN, modelNumber) ⊆ Product(ISBN, modelNumber)
  • includes(ISBN, modelNumber)
    • FK: includes(modelNumber) ⊆ InstrumentProduct(modelNumber)
    • FK: includes(ISBN) ⊆ Acessory(ISBN)
  • PASystemProduct (ISBN, modelNumber, range, type, power)
    • FK: PASystemProduct(ISBN, modelNumber) ⊆ Product(ISBN, modelNumber)
  • Acessory (ISBN, type)
